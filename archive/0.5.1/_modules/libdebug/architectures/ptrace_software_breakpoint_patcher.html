def software_breakpoint_byte_size() -> int:
    """Return the size of a software breakpoint instruction."""
    match libcontext.arch:
        case "amd64":
            return 1
        case "x86":
            return 1
        case _:
            raise ValueError(f"Unsupported architecture: {libcontext.arch}")
